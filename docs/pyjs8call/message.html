200    def __init__(self, destination=None, value=None):
201        '''Initialize message.
202
203        Args:
204            destination (bool): Callsign to send the message to, defaults to None
205            value (str): Message text to send, defaults to None
206
207        Returns:
208            pyjs8call.message: Constructed message object
209        '''
210        self.id = secrets.token_urlsafe(16)
211        self.type = Message.TX_SEND_MESSAGE
212        self.destination = destination
213        self.value = value
214        self.time = datetime.now(timezone.utc).timestamp()
215        self.timestamp = time.time()
216        self.params = {}
217        self.attributes = ['id', 'type', 'destination', 'value', 'time', 'params']
218        self.status = Message.STATUS_CREATED
219        self.raw = None
220        
221        # initialize common msg fields
222        attrs = [
223            'freq',
224            'dial',
225            'offset',
226            'call',
227            'grid',
228            'snr',
229            'from',
230            'origin',
231            'utc',
232            'cmd',
233            'text',
234            'speed',
235            'extra',
236            'messages',
237            'band_activity',
238            'call_activity'
239        ]
240
241        for attr in attrs:
242            self.set(attr, None)
243
244        # uppercase values in tx msg
245        if self.destination != None:
246            self.destination = self.destination.upper()
247        if self.value != None:
248            self.value = self.value.upper()
249            self.text = self.value

251    def set(self, attribute, value):
252        '''Set message attribute value.
253
254        Uses *setattr* internally to add attributes to the message object. Added attributes are tracked in the *attributes* attribute. Attributes are converted to lowercase for consistency.
255
256        Special attribute handling for consistency:
257            *call*: also sets *from* to the same value if *call* is not None and *from* is None
258            *from*: also sets *origin* to the same value
259
260        Note that attempting to access Message.from results in an error.
261
262        Args:
263            attribute (str): Name of attribute to set
264            value (str): Value of attribute to set
265        '''
266        attribute = attribute.lower()
267        setattr(self, attribute, value)
268
269        if attribute not in self.attributes:
270            self.attributes.append(attribute)
271
272        # set 'from' = 'call' for consistency
273        if attribute == 'call' and value != None and self.get('from') == None:
274            self.set('from', value)
275
276        # Message.from cannot be called directly, use origin instead
277        if attribute == 'from':
278            self.set('origin', value)
279
280    def get(self, attribute):
281        '''Get message attribute value.
282
283        Uses *getattr* internally.
284
285        Args:
286            attribute (str): Name of attribute to get
287
288        Returns:
289            Value of specified attribute, or None if the attribute does not exist
290        '''
291        return getattr(self, attribute, None)

293    def dict(self, exclude=[]):
294        '''Get dictionary representation of message object.
295
296        Message attributes set to None are not included in the returned dictionary.
297
298        Special attribute handling:
299            *value*: If None, set to '' (empty string); if set and Message.type is TX_SEND_MESSAGE and Message.destination is not None (i.e. directed message), set to Message.destination and Message.value joined with a space.
300
301        Args:
302            exclude (list): List of attribute names to exclude from the dictionary (see *pack*)
303
304        Returns:
305            dict: Dictionary of message attributes and values
306        '''
307        data = {}
308        for attribute in self.attributes:
309            # skip attribues excluded or already in dict
310            if attribute in exclude or attribute in data.keys():
311                continue
312
313            value = self.get(attribute)
314
315            # handle special cases
316            if attribute == 'value':
317                # replace None with empty string, 'value' is always included
318                if value == None:
319                    value = ''
320                # build directed message
321                elif self.type == Message.TX_SEND_MESSAGE and self.destination != None:
322                    value = self.destination + ' ' + value
323
324            # add to dict if value is set
325            if value != None:
326                data[attribute] = value
327
328        return data

330    def pack(self, exclude=['id', 'destination', 'time', 'from', 'origin', 'text']):
331        '''Pack message for transmission over TCP socket.
332
333        Args:
334            exclude (list): List of attribute names to exclude from the dictionary, defaults to *['id', 'destination', 'time', 'from', 'origin', 'text']*
335            
336        Returns:
337            UTF-8 encoded byte string. A dictionary representation of the message attributes is converted to a string using *json.dumps* before encoding.
338
339        '''
340        #TODO make sure 'text' is not used since it is excluded by default
341
342        data = self.dict(exclude = exclude)
343        # convert dict to json string
344        packed = json.dumps(data) + '\r\n'
345        # return bytes
346        return packed.encode('utf-8')

348    def parse(self, msg_str):
349        '''Load message string into message object.
350
351        *Message.parse* should be called inside a try/except block to catch parsing errors.
352
353        Args:
354            msg_str (str): Received message string to parse and load
355
356        Returns:
357            pyjs8call.message: Constructed messsage object
358        '''
359        self.raw = msg_str
360        msg = json.loads(msg_str)
361
362        self.type = msg['type'].strip()
363        
364        # handle inbox messages
365        if self.type == Message.MESSAGES:
366            self.messages = [m['params'] for m in msg['params']['MESSAGES']]
367            
368            for i in range(len(self.messages)):
369                self.messages[i] = {
370                    'id' : self.messages[i]['_ID'],
371                    'time' : self.messages[i]['UTC'],
372                    'origin' : self.messages[i]['FROM'],
373                    'destination' : self.messages[i]['TO'],
374                    'path' : self.messages[i]['PATH'],
375                    'text' : self.messages[i]['TEXT']
376                }
377
378        # handle call activity
379        elif self.type == Message.RX_CALL_ACTIVITY:
380            self.call_activity = []
381            for key, value in msg['params'].items():
382                if key == '_ID' or value == None:
383                    continue
384
385                call = {
386                    'origin' : key,
387                    'grid' : value['GRID'],
388                    'snr' : value['SNR'],
389                    'time' : value['UTC']
390                }
391
392                self.call_activity.append(call)
393
394        # handle band activity
395        elif self.type == Message.RX_BAND_ACTIVITY:
396            self.band_activity = []
397            for key, value in msg['params'].items():
398                try:
399                    # skip if key is not a freq offset (int)
400                    int(key)
401
402                    data = {
403                        'freq' : value['DIAL'],
404                        'offset' : value['OFFSET'],
405                        'snr' : value['SNR'],
406                        'time' : value['UTC'],
407                        'text' : value['TEXT']
408                    }
409
410                    self.band_activity.append(data)
411                except:
412                    continue
413
414        else:
415            if 'value' in msg.keys():
416                self.value = msg['value'].strip()
417
418            # parse remaining msg fields
419            for param, value in msg['params'].items():
420                param = param.strip()
421                if isinstance(value, str):
422                    value = value.strip()
423
424                self.set(param, value)
425
426            #TODO copied from js8net, test
427            if self.cmd == 'GRID' and self.text != None:
428                grid = self.text.split()
429                if len(grid) >= 4:
430                    grid = grid[3]
431                
432                if Message.ERR in grid:
433                    self.set('grid', None)
434                else:
435                    self.set('grid', grid)
436
437        # allow usage like: msg = Message().parse(rx_str)
438        return self

359        '''

362
363        self.type = msg['type'].strip()
364        
365        # handle inbox messages
366        if self.type == Message.MESSAGES:
367            self.messages = [m['params'] for m in msg['params']['MESSAGES']]
368            
369            for i in range(len(self.messages)):
370                self.messages[i] = {
371                    'id' : self.messages[i]['_ID'],
372                    'time' : self.messages[i]['UTC'],
373                    'origin' : self.messages[i]['FROM'],
374                    'destination' : self.messages[i]['TO'],
375                    'path' : self.messages[i]['PATH'],
376                    'text' : self.messages[i]['TEXT']
377                }
378

380        elif self.type == Message.RX_CALL_ACTIVITY:
381            self.call_activity = []
382            for key, value in msg['params'].items():
383                if key == '_ID' or value == None:
384                    continue
385
386                call = {
387                    'origin' : key,
388                    'grid' : value['GRID'],
389                    'snr' : value['SNR'],
390                    'time' : value['UTC']
391                }
392
393                self.call_activity.append(call)
394
395        # handle band activity
396        elif self.type == Message.RX_BAND_ACTIVITY:
397            self.band_activity = []
398            for key, value in msg['params'].items():
399                try:
400                    # skip if key is not a freq offset (int)
401                    int(key)
402
403                    data = {
404                        'freq' : value['DIAL'],
405                        'offset' : value['OFFSET'],

409                    }
410
411                    self.band_activity.append(data)
412                except:
413                    continue
414

417                self.value = msg['value'].strip()
418
419            # parse remaining msg fields
420            for param, value in msg['params'].items():
421                param = param.strip()
422                if isinstance(value, str):
423                    value = value.strip()

426
427            #TODO copied from js8net, test

431                    grid = grid[3]
432                
433                if Message.ERR in grid:
434                    self.set('grid', None)
435                else:
436                    self.set('grid', grid)
437
438        # allow usage like: msg = Message().parse(rx_str)
439        return self

  • extra (str): Used by certain JS8Call API messages, defaults to None
  • messages (list): Inbox messages, defaults to None
  • band_activity (list): JS8Call band activity items, defaults to None
